the text-indent / text-overflow / word-wrap / overflow-wrap / word-break / hyphens properties 

SYNTAX :    text-indent: cssLength                                  // defines the indentitation of the first line in a text-box (negative values are allowed)
                       : nr%                // relative to the parent element's width
                       
            text-overflow: clip             // text clipped when reaches the border         // specifies how the overflowed content is signaled to the user
                         : ellipsis         // (...) represents the clipped text 
                         : text             // set the given text to represent clipping
                         
  // the below 4 properties are doing mostly the same with different approach ----------------------------------------------------------
            word-wrap: normal               // no word break        //  long words (longer the the parent element) to be broken and wrap onto the next line (predecessor of the overflow-wrap property [does the exact same thing])  
                     : break-word           // allows unbreakable words to be broken  
            
            overflow-wrap : normal          // no word break        //  allows long words (longer the the parent element) to be broken and wrap onto the next line (new version of word-wrap property  [does the exact same thing])  
                          : break-word      // allows unbreakable words to be broken 
                          
            word-break: normal              // no word break        //  specifies how words should break when reaching the end of a line 
                      : break-all           // break word between any two characters (Prevents Overflow) 
                      : keep-all            // word break is allowed at - hyphens only! (Does NOT Prevent Overflow) 
                      : break-word          // wraps the word onto a new live if is possible otherwise breaks it (Prevents Overflow) 
            
            hyphens: none                   // no hyphenation       //  controls the text hyphenation (hyphen used when the text reaches the end of the line)
                     manual                 // (default) words are hyphenated at -(hyphen characters) 
                     auto                   // browser hyphenate when algorithm is deciding (depending on the set language) 
    
// text-indent / text-overflow -------------------------------------------------
    .myElement {
        text-indent: 50%;                                           // text indent is half comparetd to the parent element
        text-overflow: ellipsis;                                    // overflowed contend indicated by "..." (three dots)
    }
    
// word-wrap / overflow-wrap / word-break / hyphens ---------------------------- (all 4 properties seems to do the same with different approach) ---
    .myElement {
        word-wrap: break-word;                                      // long words broken to prevent overflow    // -! predecessor of "overflow-wrap" property (does the same thing)  
      // ----------------------------------
        oveflow-wrap: break-word;                                   // long words broken to prevent overflow    // -! newer version of "word-wrap" property (does the same thing) 
      // ----------------------------------
        word-break: break-all;                                      // break words where they hit the end of the line 
      // ----------------------------------
        hyphens: auto;                                              // browser alogrithm manages the hyphenation of the text (language specific) (text is broken at manually set hyphen characters for now [TESTED 2019-04-10])   
    }

the direction / unicode-bidi / writing-mode properties and dir attribute 

SYNTAX :    <HTMLTag dir="ltr/rtl">                                 // (global attribute) specifies the writing direction of an element 

            direction: ltr                  // left-to-right        // defines the writing direction in a block level element
                     : rtl                  // right-to-left
            unicode-bidi: embed                                     // used with the direction property, specifies if the text should be overridden to support multilanguage 
                        : bidi-override     // reorders the text defined by the direction property
                        
            writing-mode: horizontal-tb     // normal flow          // specifies which direction the text lines are laid out
                        : vertical-lr       // vertical, starts from the left new lines are inserted to right 
                        : vertical-rl       // vertical, starts from the right new lines are inserted to left 
    
// HTML
    <html lang="en" dir="rtl">                                      // the whole page's elements are rendered from right-to-left

// CSS
    .myElement {
        direction: rtl;                                             // element's text direction goes from right-to-left
        unicode-bidi: bidi-override;                                // writing direction is set to right-to-left (by the "direction" property)
        writing-mode: vertical-rl;                                  // text inserted vertically from right-to-left
    }

the vertical-align / line-height properties 

SYNTAX :    line-height: normal             // font size height     // specifies the height of a line 
                       : nr                 // compared the current font size (default 0) 
                       : nr%                // compared the current font size (default 100%) 
                       : cssLength          // fixed value 

            vertical-align: baseline                                // sets the vertical alignement of an inline or a table cell element (negative values are allowed)
                          : cssLength        // fixed value compared to the baseline (default 0) 
                          : nr%              // value compared to the line-height (default 0) 
                          : sub              // subscript baseline
                          : super            // superscript baseline   
                          : top              // top of the tallest element on the line
                          : text-top         // top of the parent element's font 
                          : bottom           // bottom of the lowest element on the line
                          : text-bottom      // bottom of the parent element's font 
    
    .myElement {
        line-height: 1.5;                   // line height is 1 and a half size compared to the set font-size
        line-height: 150%;                      // same effect expressed in percent
        line-heihgt: 21px;                  // line height set to a fixed value
    }

// vertical-align
    .myElement .child {
        vertical-align: 50%;                // element is 1/2 rised compared to the element line-height
        vertical-align: -25px;              // element lowered 25px compared to the base line
        vertical-align: text-top;           // element aligns to top with the tallest element in the line
    }
    .table td {
        vertical-align: bottom;             // table cell content aligned to the bottom
    }

the white-space property 

SYNTAX :    white-space: value                                      // specifies how the white-space inside an element is handeled
white-pace:value New Line Character ↵
(when carriage return encountered) 		Spaces and Tabs
(extra space between words) 		Text Wrap
(text wraps when necessary)
normal (default) Ignored Collapse Wrap
nowrap Ignored Collapse No Wrap
pre (like the <pre> tag) New Line Preserve No Wrap
pre-wrap New Line Preserve Wrap
pre-line New Line Collapse Wrap 
    .myElement {
        white-space: normal;                // carriage return ignored           /  extra white space collapsed  /  text wraps when it hits the end of the element 
        white-space: nowrap;                // carriage return ignored           /  extra white space collapsed  /  text does not wrap when it hits the end of the element 
        white-space: pre;                   // carriage return creates new line  /  extra white space preserved  /  text does not wrap when it hits the end of the element
        white-space: pre-wrap;              // carriage return creates new line  /  extra white space preserved  /  text wraps when it hits the end of the element
        white-space: pre-line;              // carriage return creates new line  /  extra white space collapsed  /  text wraps when it hits the end of the element 
    }

    }

    pre {
        tab-size: 16;                                               // tab size set to 16 characters width
        -moztab-size: 16;
        -o-tab-size: 16;

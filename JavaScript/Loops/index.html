            'prop' in obj                                                              // returns true if the property appears anywhere in the object's prototype chain, otherwise returns false 
            
            for(var of obj) {var}                                                      // iterates over an iterable object (object having the Symbol.iterator implemented) (var = current value)

                                                                                          iterates over 'real characters' in a string (characters combined by 2 UTF-16 blocks)
    
    var myObject = {Name:"Arpad", surename:"Pall", age:32};
    var newObject = Object.create(myObject, {});                                       // the 'myObject' is the prototype of thi object 
    var myArray = ["Arpad", "Pall", 32];
    
// for ...in --------------------------------------------------------------------------
    for (let i in myObject){                                                           // iterating over an object's properties 
        i;                                                                             // -> Name | surename | age      // properties are returned
    }
    
    for (let i in myArray) {                                                           // iterating over an Array 
        i;                                                                             // -> 1 | 2 | 3                  // indexes are returned (this loop is not used for iterate over Arrays!)   
    }
    
    'Name' in myObject                                                                 // -> true                       // native property of the object 
    'Name' in newObject                                                                // -> true                       // property appears in the object's prototype chain   
    'toString' in newObject                                                            // -> true                       // property appears in the object's prototype chain   
    
// for ...of --------------------------------------------------------------------------
    for (let i of myArray) {                                                           // iterating over an Array's items 
        i;                                                                             // -> "Arpad" | "Pall" | 32      // items are returned in the sequence 
    }  
    
    for(let i of 'lo ðŸ´'){                                                            // this loop iterates over any objects or values that has the [Symbol.iterator] implemented (like strings)
        i;                                                                             // ->  'l'| 'o' | ' ' | 'ðŸ´'     // the 'ðŸ´' character is composed of 4 bytes (2 x 2 byte block) but this loop can handle this

the while and do...while loops

- the while loop loops through a block of code until the condition is true 

SYNTAX:     while(cond) { }                                                            // if the condition evaluates true runs through the block of code {}, then checks the condition again    
            
            do { } while(cond)                                                         // runs through the block of code once, then if the condition evaluates true runs through the block of code, then checks the condition again    
    
    let i = 0;
    
// while ------------------------------------------------------------------------------
    while (i < 5){                                                                     // checks the condition before every pass 
        i++                                                                            // code insdie the block runs 5 times 
    } 
    
// do...white ------------------------------------------------------------------------- 
    do {
        i++                                                                            // -! the first pass is garanteed to run before the condition is checked    
    } while(i < 5);                                                                    // checks the condition before every pass (except the first pass)

using break // continue // label statements in loops

- the label statement provides an identifier for a loop, these labels only usable by break and continue statements to identify whitch statement to apply

- the break statement (can break any {} code blocks), prevents all further code to be executed in the current block and blocks the loop

- the continue statement (only usable by loops), prevents all further code to be executed in the current block but it desn't block the loop 

SYNTAX :    labelName :                  // labels the loop   
            for (;;){
                break labelName          // if a label is declared the break is valid on that loop      
            }
            
            labelName :                  
            for (;;){
                continue labelName       // if a label is declared the continue is valid on that loop      
            }
    
// break -------------------------------------
    for (let i = 0; i < 3; i++) {               // 3 passes
        document.write("before break");
        break;                                  
        document.write("after break");          // further code is not executed and the loop is blocked
    }
    // -> "before break"                        // printed only once
    
// continue ----------------------------------
    for (let i = 0; i < 3; i++) {               // 3 passes
        document.write("before continue");
        continue;                                
        document.write("after continue");       // further code is not executed
    }
    // -> "before continue" / "before continue" / "before continue"     // the code before the continue statement is printed every time but the code after it is not!
    
    //  --------------------------------------
    let i = 0;
    while (i < 3) {
        document.write("before continue");
        continue;
        i++;                                    <- this creates an infinite loop because the code after the "continue" statement is skipped every time!
    }

- labels 

 
    loop1:
    for (i = 0; i < 3; i++) {                   // The first for statement is labeled "loop1"
       loop2:
       for (j = 0; j < 3; j++) {                // The second for statement is labeled "loop2"
          if (i === 1 && j === 1) {
             break loop1;                       // parent loop "loop1" will break when condition is true
          }
          console.log('i = ' + i + ', j = ' + j);
       }
    }

Remember This :

- avoid creating arrays with the new keyword, it complicates your code and slows down execution speed

- the indexOf() searches after the set index and the lastIndexOf() searches before set index

- the forEach() method has better performance over the map() method because it does not return anything (but it is situational which to use)

Description and Demonstration :

- a JavaScript Array is a special variable, which can hold more than one value at the time and maintains their order in the array 

SYNTAX :    [item, item, ...]                                   // Array literal, creates an array and returns with the passed items       
            new Array(item, ...)                                // creates an array with the passed items (avoid it slows execution)
            new Array(nr)                                       // creates an array empty Array with a set length (avoid it slows execution)

    ["string", 21, {prop:true},];                               // -! coma tailing is forbiden in IE8 and earlier

Array items

- we can basically store any valid data in an array even functions 

    var myArray = [ "Pall",                                     // String
                    21,                                         // Number
                    false,                                      // Boolean
                    ["Bircsy", 33, false],                      // Array Object
                    myFunction,                                 // function Object
                    {'1st':"Me", '2nd':"You", '3rd':"he"}       // Object
    ];
    
    function myFunction(a, b) { return a + b; };
    
// calling stored items ----------------------------------------- 
    nyArray[1];                                                 // -> 21
    myArray[3][0];                                              // -> "Bircsy"      // accessing an Array in the Array
    myArray[4](3, 4);                                           // -> 7             // call a function stored in Array 
    myArray[5].2nd;                                             // -> "You"         // accessing an Object in the Array
    
// create new or replace existing items ------------------------- 
    myArray[3] = "Array Replaced";                              // 4th element is replaced 
    myArray[6] = "new item";                                    // new item added at the end    // -! this creates a hole for myArray[5] item
    
// remove items ------------------------------------------------- 
    delete myArray[5];
    myArray[5] = undefined;                                     // this would do too

Arrays VS Objects

- JavaScript Array is a sequentian data structure elements are stored in a linear way, so we acces items by refering to the order number "index" associated with the item (Array order starts with 0)

- JavaScript Object stores items by using "key/value" pairs, so we acces items by refering to the key associated with the item (item order is NOT maintainerd!)

- at end Arrays are also an Objects, the just offer different ways to deal with the data 

    var this_array = ["Arpad", "Pall"];                             
    var this_object= {firstName:"Arpad", lastName:"Pall"};
    
    this_array[0];                                              // -> "Arpad"       // accessing Array item
    this_object.firstName;                                      // -> "Arpad"       // accessing Ojbect item
    
    typeof this_array                                           // -> "object"      // -! any non primitive data is returned as object in JavaScript (even null)

Properties and Methods

Static methods (Array() constructor methods) isArray() / from() / of() 

SYNTAX :    Array.isArray(value)                                // returns a boolean value indicating if the passed value is an Array

            Array.from(iterableObj, mapFn(value, index, array), thisValue)      // 6 returns a new shallow copied Array from the passed iterable object   
                - iterableObj : any iterable object
                - mapFn(value, index, array) : map function used on the iterable objects items
                - thisValue : passed "this" value used in the mapFn(value, index, array) function
                
            Array.of(val, val, ...)                             // 6 creates a new Array with the passed values regardless of their data type (avoid pitfalls when creating an Array by using the "Array()" constructor)  
    
// isArray() ---------------------------------------------------------------
    Array.isArray([1, 3]);                                      // -> true
    Array.isArray({1:3})                                        // -> false
    Array.isArray("myString")                                   // -> false
    
// from() ------------------------------------------------------------------
    Array.from("Pall");                                         // -> ["P", "a", "l", "l"]      // string has the "length" property 
    Array.from(new Set([4, false, "myTxt"]));                   // -> [4, false, "myTxt"]       // "Set" is an iterable object
    Array.from(["egy", "ketto", "harom"], (val, ind) => val = val + " ind = " + ind);   // -> ["egy ind = 0", "ketto ind = 1", "harom ind = 2"]   // callback function used to handle passed values   
    Array.from({prop1:"val", prop2:"val", prop3:[1, 2, 4]})     // -> []                        // -! objects does not have the "length" property therefore the new array cannot be created! 
    
    // creates a new Array (a new copy) -------------- 
    var myTestObj_1 = [1, 2, ["a", "b", "c"]];
    var myTestObj_2 = Array.from(myTestObj_1);                  // -! a copy is created form the original Array in this situation 
    myTestObj_1[1] = false;                                     // we modify the first Array 
    delete myTestObj_1[2];
    
    myTestObj_2;                                                // -> [1, 2, ["a", "b", "c"]]   // the newly created Array is a new copy (never gets muted)
    
// of() --------------------------------------------------------------------
    new Array(7);                                               // -> [,,,,,,,]     // an Array with 7 empty values
    Array.of(7);                                                // -> [7]           // returns an Array with a value (numbers can be safely passed)

The pop() / push() / shift() / unshift() methods 

SYNTAX:     array.pop()                                                                // removes the last item of an Array and returns it
            array.push(intem, item, ...)                                               // adds item(s) at the end of the Array and returns the new Array length    
                
            array.shift()                                                              // removes the first item of an Array and returns it    
            array.unshift(intem, item, ...)                                            // adds item(s) at the beginning of the Array and returns the new Array length
DEMO 
    var myArray  = ["Break", "Berlin", "Citadine", "SUV"]; 
    
// pop() ------------------------------------------------------------------------------ 
    myArray.pop();                                                                     // -> "SUV"             // last item is removed and returned 
    myArray;                                                                           // -> ["Break", "Berlin", "Citadine"]
    
// push() -----------------------------------------------------------------------------   
    myArray.push("Convertable", "Sport");                                              // -> 5                 // new items are pushed at the END in the Array and new length is returned    
    myArray;                                                                           // -> ["Break", "Berlin", "Citadine", "SUV", "Convertable", "Sport"]    
    
// shift() ----------------------------------------------------------------------------  
    myArray.shift();                                                                   // -> "Break"           // first item is removed and returned 
    myArray;                                                                           // -> ["Berlin", "Citadine", "SUV"]     
    
// unshift() -------------------------------------------------------------------------- 
    myArray.unshift("Kombi");                                                          // -> 5                 // inserts the passed item at the beginning of the Array and returns the new Array length    
    myArray;                                                                           // -> ["Kombi", "Break", "Berlin", "Citadine", "SUV"]

- splice() / slice() / concat() / flat() methods 

SYNTAX:     array.splice(indexPosition, NR_howmanyRemove, item, item, ...)      // add/removes items to/from the Array and returns the removed elements in a new Array (original Array is modified)
                - indexPosition : position to perform in
                - NR_howmanyRemove : how many items to remove
                - item1, item2, ... : items to be added in Array
                
            array.slice(stIndex(included), endIndex(excluded))  // returns a new Array from the selected items (original Array is not modified)
            array.concat(Array/item, ...)                       // merges Arrays together or merges the passed items into the source Array and returns a new Array (original Array is not modified)
            
            array.flat(depth_nr)                                // 10 concatenates nested-arrays into the original Array and returns a new Array (works only with nested Arrays (Objects are ignored), removes empty items) (original Array is not modified)  
                - depth_nr : defines how deep a nested Array should be flattened (default 1)
    
    var myArray = ["Audi", "BMW", "Mercedes", "Dacia", "Toyota"];
    var myMultiArray = [1, false,  ,"string", {p1:"val", p2:[1, 3]}, ["egy", "ketto", ["un", "deux", ["one", "two"]]]];     // there is an empty item in this Array!
    
// splice() --------------------------------------------------------------------  
    myArray.splice(1, 0, "Opel", "Ferrary");                    // -> []                    // 2 new items are added, nothing removed 
    myArray;                                                    // -> ["Audi", "Opel", "Ferrary", "BMW", "Mercedes", "Dacia", "Toyota"]
    myArray.splice(2, 2);                                       // -> ["Ferrary", "BMW"]    // 2 items are removed and returned 
    myArray;                                                    // -> ["Audi", "Opel", "Mercedes", "Dacia", "Toyota"]
    myArray.splice(3, 1, "Renault");                            // -> ["Dacia"]             // 1 item replaced 
    myArray;                                                    // -> ["Audi", "Opel", "Mercedes", "Renault", "Toyota"] // -! original Array is modified
        
// slice() --------------------------------------------------------------------- 
    myArray.slice();                                            // -> ["Audi", "BMW", "Mercedes", "Dacia", "Toyota"]    // copy of the original array
    myArray.slice(1, 3);                                        // -> ["BMW", "Mecedess"]
    myArray;                                                    // -> ["Audi", "BMW", "Mercedes", "Dacia", "Toyota"]    // !! original Array is not modified  
    
// concat() -------------------------------------------------------------------- 
    myArray.concat(["Renault", "Citroen"]);                     // -> ["Audi", "BMW", "Mercedes", "Dacia", "Toyota", "Renault", "Citroen"]  // Array merged together and returns a new Array 
    myArray.concat("Renault", "Citroen");                       // -> ["Audi", "BMW", "Mercedes", "Dacia", "Toyota", "Renault", "Citroen"]  // merges the passed values into the source Array and returns a new Array   
    myArray;                                                    // -> ["Audi", "BMW", "Mercedes", "Dacia", "Toyota"]    // !! original Array is not modified
    
// flat() ----------------------------------------------------------------------  
    // -! flattens nested Arrays Only! and removes empty items 
    myMultiArray.flat();                                        // -> [1, false, "string", {p1:"val", p2:[1, 3]}, "egy", "ketto", ["un", "deux", ["one", "two"]]]           // 1 depth (default)
    myMultiArray.flat(2);                                       // -> [1, false, "string", {p1:"val", p2:[1, 3]}, "egy", "ketto", "un", "deux", ["one", "two"]]             // 2 depth 
    myMultiArray.flat(3);                                       // -> [1, false, "string", {p1:"val", p2:[1, 3]}, "egy", "ketto", "un", "deux", "one", "two"]               // 3 depth
    myMultiArray;                                               // -> [1, false,  ,"string", {p1:"val", p2:[1, 3]}, ["egy", "ketto", ["un", "deux", ["one", "two"]]]]       // -! original Array is not modified

SYNTAX :    array.indexOf(item, index(included))                // searches for an item and returns the first found item's index (if the index is specified the the array is checked after set index (negative index allowed)) (-1 indicates not match)

    myArray.indexOf("BMW", -2);                                 // -> -1            // search begins from the 5th index(included) in this situation (item not found)
    
// lastIndexOf() ---------------------------------------------------------------  
    myArray.lastIndexOf("BMW");                                 // -> 4             // search starts from the end-to-start (first item found index returned)
    myArray.lastIndexOf("BMW", 3);                              // -> 1             // search starts from the index 3 to the start 
    
// included() ------------------------------------------------------------------  
    myArray.includes("BMW");                                    // -> true          
    myArray.includes("BMW", -2);                                // -> false         // search begins from the 5th index(included) in this situation (item not found)

SYNTAX:     array.copyWithin(indexPaste, stIndex(included), endIndex(excluded))    // 6 copies items within the Array to the targeted index (replaces items) (cannot extend Array size)
                - indexPaste : Index position to paste elements 
                - stIndex(included) : The Index position to start copying elements form (default 0)
                - endIndex(excluded) : The Index position to stop copying elements form (if omitted all elements are copyed after "start")
            
            array.fill(item, stIndex(included), endIndex(excluded))                // 6 fills the targeted index range with the passed value (all items are replaced with the passed item) (by default replaces all items) (cannot extend Array size)

    myArray;                                                    // -> ["Audi", "BMW", " - ", " - ", "BMW", "Toyota", "Renault"]             // original Array is modified
    myArray.fill("extend", -1);                                 // -! original Array length cannon be extended with this property
    myArray;                                                    // -> ["Audi", "BMW", " - ", " - ", "BMW", "Toyota", "extend"]

- the forEach() method has better performance over the map() method because it does not return anything (but it is situational which to use) 

SYNTAX:     array.forEach(fn(value, index, array), thisValue)   // runs the callback function for each element in the Array but cannot return anything nor modifies the original Array (the callback function have ONLY side effect, "return" has no effect)    
            array.map(fn(value, index, array), thisValue)       // runs the callback function for each element in the Array and returns the result in a new Array (original Array is not modified)   
                - fn : callback function which runs for every element
                    - value : curren item
                    - index : current item index
                    - array : the Array itself
                - thisValue : value what can be used as "this" value in the callback function 
    
    var myArray = [10, 20, 30];
// map() -----------------------------------------------------------------------    
    myArray.map(function(val, index) {                          // -> [11, 21, 31]      // new Array returned
        return val +=1;
    });
    myArray;                                                    // -> [10, 20, 30]      // original Array is not modified  
    
// forEach() -------------------------------------------------------------------   
    forEach(function(val, index) {                              // -> undefined         // -! the "forEach" method does not return values!
       // do something with the passed values                   // the "forEach" method is designed to have ONLY side effects
       return val +=1;                                          // -! this doesn't do anything because the "return" has no effect in the "forEach" callback function
    });
    myArray;                                                    // -> [10, 20, 30]      // original Array is not modified

- filter() / find() / findIndex() / every() / some() methods 

SYNTAX :    array.filter(fn(value, index, array), thisValue)    // returns all found elements provided by the callback function in a new Array 
            array.find(fn(value, index, array), thisValue)      // 6 returns the first found item provided by the callback function (than stop calling the callback function further)   
            array.findIndex(fn(value, index, array), thisValue) // 6 returns the index of the first item that passes the text provided by the callback function (than stop calling the callback function further)   
            
            array.every(fn(value, index, array), thisValue)     // returns "true" if all items passes a test provided by the callback function, otherwise returns "false"
            array.some(fn(value, index, array), thisValue)      // retruns "true" if at least one item passes a test provided by the callback function, "false" if there is no match (if one item passes the test the callback function is not called any further)   
               
                - fn : callback function which runs for every element
                    - value : curren item
                    - index : current item index
                    - array : the Array itself
                - thisValue : value what can be used as "this" value in the callback function 
    
    var myArray = ["Audi", "BMW", "Mercedes", "Dacia", "BMW", "Toyota", "Renault"];
    
// filter() -------------------------------------------------------------------- 
    myArray.filter((val, index, arr) => {                       // -> ["Mercedes", "Dacia", "Toyota", "Renault"]     // new Array returned 
        if (val.length >= 5) return val;                        // we filter all items what is equal or longer than 5 characters 
    });
    
// find() ----------------------------------------------------------------------  
    myArray.find((val, index, arr) => {                         // -> "BMW"
        if (index >= 2 && /W$/.exec(val)) return val;           // we filter for the first match which is after index 2(included) and ends with the letter "W"
    });
    
// findIndex() ----------------------------------------------------------------- 
    myArray.findIndex((val, ind, arr) => {                      // -> 1
        return val == "BMW";                                    // index returned for the first item which passes the test 
    });
  
// every() --------------------------------------------------------------------- 
    myArray.every((val, index, arr) => {                        // -> true 
        return typeof val == "string";                          // checks if all items are strings 
    });
             
// some() ----------------------------------------------------------------------  
    var result = myArray.some((val, index, arr) => {            // -> false
        return typeof val == "number";                          // checks if there is any number in the Array
    });

- reduce() / reduceRight() methods 

SYNTAX :    array.reduce(fn(total, value, index, array), thisValue)        // reduces the Array to a single value by running a callback function for every element (start-to-end) and accumulating them in a single value (does not modify the original Array)    
            array.reduceRight(fn(total, value, index, array), thisValue)   // reduces the Array to a single value by running a callback function for every element (end-to-start) and accumulating them in a single value (does not modify the original Array)   
            
             - fn() : callback function which runs for every element
                - total : value previously returned by the callback function (in the first pass it is the thisValue if specified, otherwise it is the first/last item of the array) 
                - value : current value (in the first pass if the thisValue is specified it is the second item of the Array)  
                - index : current index of the element
                - array : the Array object itself
            - initialValue : is the total in the first pass
    
    var myArray = ["egy", "ketto", "harom"];
    
// reduce() --------------------------------------------------------------------  
    myArray.reduce(function(total, val, ind, arr){              // -> "egyketto harom"          // callback function runs 2 times 
        return total + val + " ";                               // in the first pass (total = "egy") (val = "ketto")
    });
    myArray.reduce(function(total, val, ind, arr){              // -> "-> egy ketto harom"      // callback function runs 3 times 
        return total + val + " ";                               // in the first pass (total = "-> ") (val = "egy")
    }, "-> ");                                                  // -! the "this" value is the first item in the first pass
    
// reduceRight() ---------------------------------------------------------------  
    reduceRight(function(total, val, ind, arr){                 // -> "haromketto egy"          // callback function runs (end-to-start) 2 times 
        return total + val + " ";                               // in the first pass (total = "harom") (val = "ketto")
    });
    
    myArray.reduceRight(function(total, val, ind, arr){         // -> "-> harom ketto egy"      // callback function runs (end-to-start) 3 times 
        return total + val + " ";                               // in the first pass (total = "-> ") (val = "harom")
    }, "-> ");                                                  // -! the "this" value is the first item in the first pass

- sort() / reverse() methods 

SYNTAX:     array.sort(compareFn(a, b))                         // sorts the Array items in ascending order according to the Unicode Table (optionally uses a callback function to a special sort) (modifies the original Array)   
                - compareFn(a, b) : converts the passed items to numbers and compares them against each other if returned value is (negative a is lower than b) (positive a is higher than b) (0 a and b are equals) 
                
            array.reverse()                                     // reverses the Array order (modifies the original Array)

- the compareFn(a, b) (callback compare function) compares each Array items together by converting them into a number

- (if returned value is negative than a is lower than b)   (if the returned value is a positive than a is higher than b)   (if the returned value is 0 than a and b are equals)

- if any of the passed items cannot be converted to a number the Array order is not changed!

    var myArray_str = ["Bravo", 41, "Charlie", 4, 12, "Alpha"]; 
    var myArray_nr = [40, 100, 1, 5, 25, 10];
    var myArray_bol = [false, "34", 2, "8", true, 23, 1024];
    
// sort() ----------------------------------------------------------------------  
  // !! default sort order without the compareFunction is the sequence of UTF-16 code unit values  
    myArray_str.sort();                                         // -> [12, 4, 41, "Alpha", "Bravo", "Charlie"]      // returns the sorted Array
    myArray_nr.sort();                                          // -> [1, 10, 100, 25, 40, 5]                       // -! does not work very well with numbers! 
    myArray_bol.sort();                                         // -> [1024, 2, 23, "34", "8", false, true]         
    
  // use compare function ------------------------------------------------------  
    // -! the compare callback function is designed to sort numbers 
    myArray_str.sort((a, b) => {                                // -> ["Bravo", 41, "Charlie", 4, 12, "Alpha"]      // no changes because the Array contains strings
        return a - b;                                           // -! because the strings in the Array are converted to NaN we cannot calculate the Array order 
    });
    myArray_nr.sort((a, b) => {                                 // -> [1, 5, 10, 25, 40, 100]                       // numbers are sorted correctly
        return a - b;                                           // each value is compared against each other (explainded above) 
    });
    myArray_bol.sort((a, b) => {                                // -> [false, true, 2, "8", 23, "34", 1024]         // sorted correctly 
        return a - b;                                           // -! because every element can be converted to a number we can sort this  
    });
    
// reverse() -------------------------------------------------------------------  
    myArray_str.reverse();                                      // -> ["Alpha", 12, 4, "Charlie", 41, "Bravo"]      // returns the reversed Array
    myArray_nr.reverse();                                       // -> [10, 25, 5, 1, 100, 40]                       
    myArray_bol.reverse();                                      // -> [1024, 23, true, "8", 2, "34", false]

- entries() / keys() methods 

SYNTAX :    array.keys()                                        // 6 returns an Iterable Object [Iterable Objects (link)] with the keys (indexes) of each item 
            array.entries()                                     // 6 returns an Iterable Object [Iterable Objects (link)] with the key(index)/value pairs of each item 
    
    var myArray = [23, "myString", {prop1:"val", prop2:"val"}, [1, 2, 3]];
    
// keys() ----------------------------------------------------------------------  
    var iterObj = myArray.keys();                               // iterable object returned with the Array's keys (indexes)
    iterObj.next();                                             // -> {value:0, done:false}             // call items from the iterable Object 
    iterObj.next();                                             // -> {value:1, done:false}
    iterObj.next();                                             // -> {value:2, done:false}
    iterObj.next();                                             // -> {value:3, done:false}
    iterObj.next();                                             // -> {value:undefined, done:true}      // end of the iterable object 
    
// entries() ------------------------------------------------------------------- 
    var iterObj = myArray.entries();                            // iterable object returned with the Array's key(index)/value pairs 
    iterObj.next();                                             // -> {value:[0, 23], done:false}       // call items from the iterable Object 
    iterObj.next();                                             // -> {value:[1, "myString"], done:false}
    iterObj.next();                                             // -> {value:[2, {prop1:"val", prop2:"val"}], done:false}
    iterObj.next();                                             // -> {value:[3, [1, 2, 3]], done:false}
    iterObj.next();                                             // -> {value:undefined, done:true}      // end of the iterable object

SYNTAX :    element.innerHTML                                                          // gets or sets the element's content as a string (parses the string into HTML code when set)   
            element.outerHTML                                                          // gets or sets the element itself and its content as a string (parses the string into HTML code when set)   
            
            node.textContent                                                           // returns the element's and all its children elements' text content in one string (handles hidden elements as well)    
                                                                                          when set replaces all descendent nodes by the set text node     
            element.innerText                                                          // returns the element's and all its children elements' text content in one string (rendered element only, if the element is hidden by CSS this property cannot see the element but it stil can set it!)    
                                                                                          when set replaces all descendent nodes by the set text node     
            
            document.write(exp, ...)                                                   // writes the expression(s) to the document on the current line (if it is used after the document is loaded, this will delete all existing HTML)    
                                                                                          parses the string into HTML code    
            document.writeln(exp, ...)                                                 // writes the expression(s) to the document on the current line and inserts a \n (new line) character after the statement automatically (if it is used after the document is loaded, this will delete all existing HTML)    
                                                                                          parses the string into HTML code    
    
    <p id="demo"> myParagraph <u>underlined</u> </p>
    <p id="demo2" style="visibility:hidden;"> myParagraph <span>mySpan</span> myParagraph </p>
<script>
    var myElement = document.getElementById('demo');
    var myElement2 = document.getElementById('demo2');
    
// innerHTML --------------------------------------------------------------------------
    myElement.innerHTML;                                                               // -> ' myParagraph <u>underlined</u> '                    // returns element's content in a string   
    myElement.innerHTML = "<mark>marked</mark>";                                       // element looks like this =  <p id="demo"><mark>marked</mark></p>
    
// outerHTML --------------------------------------------------------------------------
    myElement.outerHTML;                                                               // -> '<p id="demo"> myParagraph <u>underlined</u> </p>'   // returns element itself and it's content in a string 
    myElement.outerHTML = "<mark>marked</mark>";                                       // element looks like this =  <mark>marked</mark>
    
// textContent / innerText ------------------------------------------------------------
    myElement2.textContent;                                                            // ->  "myParagraph mySpan myParagraph"      // the content of each descendent element is returned 
    myElement2.innerText;                                                              // ->  ""                                    // this method cannot see the element because it's hidden by CSS (otherwise would do the same as the method above)    
    
    myElement2.textContent = "replaced content";                                       // element looks like this = <p id="demo2" style="visibility:hidden;">replaced content</p>
    myElement2.innerText = "replaced content";                                         // element looks like this = <p id="demo2" style="visibility:hidden;">replaced content</p>   // -! no matter if the element is hidden we stil can set its content    
    
// write() / writeln() ----------------------------------------------------------------
    document.write('<p>', 200+2, '</p>');                                              // returns and renders this on the current line =  <p>202</p>
    document.writeln('<p>', 200+2, '</p>');                                            // returns and renders this on the current line =  <p>202</p>\n       // new line character automatically added at the end of the statement  
</script>

- the printed results by these methods are rendered as inline elements in HTML, in which case the \n (new line character) is ignored, so we wrap the printed result in <pre> like element which renders new line characters

    var writeFn = function(a, b) {
        return a**b;
    }
    
// write() ---------------------------------------------------------------------------------
    document.write('calculation result is = ', writeFn(4, 4));
    document.write('calculation result is = ', writeFn(4, 4));
    document.write('calculation result is = ', writeFn(4, 4));
    document.write('<br>');                                                 // parses the string into HTML code 
    
// writeln() -------------------------------------------------------------------------------
    document.writeln('calculation result is = ', writeFn(4, 4));
    document.writeln('calculation result is = ', writeFn(4, 4));
    document.writeln('calculation result is = ', writeFn(4, 4));
    
// string to HTML code ---------------------------------------------------------------------
    document.writeln('<ol><li>item</li><li>item</li><li>item</li></ol>');
    document.write('<ol><li>item</li><li>item</li><li>item</li></ol>');

Text and Comment Node 

SYNTAX :    textNode|commentNode.length                                                // returns the text length of the node      
            textNode|commentNode.data                                                  // gets or sets the nodes text   
            textNode|commentNode.substirngData(st(inc):index, howMany:nr)              // returns a new string from the node's text portion    
            
            textNode|commentNode.insertData(st(inc):index, str)                        // inserts a text in the node (before the index) by shifting the characters (stretches the content, does not replace characters)    
            textNode|commentNode.replaceData(st(inc):index, howMany:nr, str)           // replaces a part of the node's text with the passed string    
            textNode|commentNode.appendData(str)                                       // adds a text at the end of the node's text    
            
            textNode|commentNode.deleteData(st(inc):index, howMany:nr)                 // deletes a part of the node's text       
    
    <p id="text">text node</p><!--comment node-->
<script>
    var textNode = document.getElementById('text').childNodes[0];
    var commentNode = document.getElementById('text').nextSibling;
    
// data / length / substringData() ----------------------------------------------------
    textNode.length;                                                                   // -> 9                          
    
    textNode.data;                                                                     // -> 'text node'        
    textNode.data = 'modified node';                                                   // sets the node's value (text)  
    
    commentNode.substringData(0, 7);                                                   // -> 'comment'          // new string returned from the targeted part (7 characters from index 0)   
    
// insertData() -----------------------------------------------------------------------
    textNode.insertData(4, ' -');                                                      // inserts the passed string before the 4th index (stretches the node)   
        textNode.data;                                                                 // -> 'text - node'
    
// replaceData() -----------------------------------------------------------------------
    textNode.replaceData(4, 1, ' - ');                                                 // replaces 1 character at 4th index with the passed string (stretches the node in this case)   
        textNode.data;                                                                 // -> 'text - node'
    
// appendData() -----------------------------------------------------------------------
    textNode.appendData('!');                                                          // appends the passed string at the end of the node   
        textNode.data;                                                                 // -> 'text - node'
        
// deleteData() -----------------------------------------------------------------------
    textNode.deleteData(4, 5);                                                         // deletes 5 characters from index 4
        textNode.data;                                                                 // -> 'text'
</script>

SYNTAX :    element.tagName                                                            // returns the element's tag name in a string
            node.nodeValue                                                             // gets or sets the node's value (ex: text content of a text node)
            node.nodeName                                                              // returns name of the node in a string     
            node.nodeType                                                              // returns a number referring to the node's type (1 = element | 2 = attribute[depricated] | 3 = text | 4 = CDATA | 8 = comment | 9 = document | 10 = document type | 11 = document fragment)

    var myElement = document.getElementById("innerDemo");
    
    myElement.tagName;                                                                 // -> "P"
    myElement.nodeValue;                                                               // -> null      // the element's value as a node!
    myElement.nodeName;                                                                // -> P         // paragraph Node (element)
    myElement.nodeType;                                                                // -> 1         // 1 = refers to an element
    
    myElement.childNodes[0].nodeValue;                                                 // -> demoDemo  // the text node's value
    myElement.childNodes[0].nodeName;                                                  // -> #text     // text node
    myElement.childNodes[0].nodeType;                                                  // -> 3         // 3 = refers to a text node
</script>

Element Styling 

SYNTAX :    element.style                                                              // returns the CSSStyleDeclaration object which holds the styling information of the element (inline styles returned only!)    
                                                                                          the returned CSSStyleDeclaration get or set inline styles   
                                                                                          
            window.getComputedStyle(element, pseudoElement)                            // returns the CSSStyleDeclaration object which holds the styling information of the element 
                                                                                          the returned CSSStyleDeclaration object is read-only in this situation  
    
    <p id="demo"> demo paragraph </p>
<script>  
    var demoElement = document.getElementById('demo');
    
// style ------------------------------------------------------------------------------
    var cssInline = demoElement.style;                                                 // -> CSSStyleDeclaration {...}      // -! the returned object holds inline styles only!   
    cssInline.color = 'orange';                                                        // object can be used to set inline styles 
    
// getComputedstyle() -----------------------------------------------------------------
    var cssAll = window.getComputedStyle(demoElement);                                 // -> CSSStyleDeclaration {...}      // the returned object holds all element styles (head, external and inline styles)    
    cssAll.padding = '10px';                                                           // -! we cannot set styles in this situation (throws a DOMExeption error)   
</script>

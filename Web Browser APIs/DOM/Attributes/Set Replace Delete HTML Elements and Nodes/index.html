The createElement() / createTextNode() / createAttribute() / createCommend() methods 

SYNTAX :    document.createElement("htmlTag")                                          // creates an HTML element and returns it
            document.createTextNode("value")                                           // creates a text Node and returns it
            document.createAttribute("attribute")                                      // creates an attribute Node and returns it
            document.createComment("commentValue")                                     // creates a comment Node and returns it
    
    <div id="demo-parent">
        <p id="demo"> HTML Paragraph </p>
    </div>
<script>
    var parent = document.getElementById('demo-parent');                               // load parent in a variable 
    var element = document.getElementById("demo");                                     // load element in a variable
    
// createAttribute() ------------------------------------------------------------------    
    var attributeNode = document.createAttribute('style');                             // create the "style" attributeNode
    attributeNode.value = "text-decoration:underline;";                                // adding value to an attribute Node
    element.setAttributeNode(attributeNode);                                           // add the attribute Node to the element    = HTML Paragraph
    
// createElement() / createTextNode() -------------------------------------------------    
    var element_new = document.createElement("p");                                     // create a new paragraph Node
    var text_new = document.createTextNode("JavaScript Paragraph");                    // create a new text Node
    element_new.appendChild(text_new);                                                 // add the text Node as element's child
    parent.appendChild(element_new);                                                   // add the newly created Node as element's child (added as last element)
    
// createComment() --------------------------------------------------------------------
    var myCommentNode = document.createComment("myComment");                           // comment node created with its value
    document.getElementById("demo-parent").insertBefore(myCommentNode, element);       // comment node inserted before the targeted element
</script>

The appendChild() / append() / insertBefore() / prepend() / before() / after() / remove() / removeChild() / replaceChild() / replaceWith() methods 

SYNTAX :    parentNode.appendChild(node)                                               // appends the passed node as last child of the parentNode, returns the appended node
            parentNode.append(node|domString, ...)                                     // appends the passed node(s) as last child of the parent node  
            parentNode.prepend(node|domString, ...)                                    // appends the passed node(s) as first child of the parent node
            
            parentNode.insertBefore(newNode, referenceNode)                            // inserts the new node before the reference node and returns it    
            node.before(node|domString, ...)                                           // inserts the passed node(s) (arguments) right before the node
            node.after(node|domString, ...)                                            // inserts the passed node(s) (arguments) right after the node
            
            parentNode.removeChild(childNode)                                          // removes and returns the parent's child Node
            node.remove()                                                              // removes the node (does not return it)   
            
            parentNode.replaceChild(newNode, oldNode)                                  // replaces the parent's child node by a new node, returns the old node     
            node.replaceWith(node|domString, ...)                                      // replaces the node by the passed node(s) (arguments)   
    
    <div id="parent">
        <p id="child1"> paragraph 1 </p>
        <p id="child2"> paragraph 2 </p>
        <p id="child3"> paragraph 3 </p>
    </div>
<script>
        var parent = document.getElementById('parent');
        var chidl1 = document.getElementById('child1');
        var chidl2 = document.getElementById('child2');
        var chidl3 = document.getElementById('child3');
        
        var element1 = document.createElement('p');                                    // create a new element with text content in it 
            var textNode = document.createTextNode('paragraph X');
            element1.appendChild(textNode);
        var element2 = element1.cloneNode(true);                                       // clone other elements
        var element3 = element1.cloneNode(true);
        var element4 = element1.cloneNode(true);
    
// appendChild() / append() / prepend() / ---------------------------------------------
        parent.appendChild(element1);                                                  // -> <p>paragraph X</p>   // appends the passed element as last child of the parent and returns it   
        parent.append(element2, 'some DOMString');                                     // appends the created element and DOMString as last child of the parent (does not return it)   
        parent.prepend(element3, 'some DOMString');                                    // appends the created element and DOMString as first child of the parent (does not return it)   
        
// insertBefore() / before() / after() ------------------------------------------------
        parent.insertBefore(element1, chidl2);                                         // -> <p>paragraph X</p>   // insert the created element before the second child and returns it 
        child2.before(element2);                                                       // inserts the created element before the second child (does not return it)
        child2.after(element3);                                                        // inserts the created element after the second child (does not return it)
        
// removeChild() / remove() -----------------------------------------------------------
        parent.removeChild(child2);                                                    // -> <p> paragraph 2 </p> // removes and returns the targeted child element    
        child3.remove();                                                               // removes the the element (does not return it)   
        
// replaceChild() / replaceWith() -----------------------------------------------------
        parent.replaceChild(element1, child2);                                         // -> <p"> paragraph 2 </p> // replaces the parent's second child with the new element and returns the replaced element    
        child3.replaceWith(element2);                                                  // replaces the 3rd child element with the created element (does not return the old element)    
</script>

    var refChild = document.getElementById('insertNode');

    var tx = document.createTextNode(' JavaScript created paragraph ');

    refParent.insertBefore(par_2, refChild.nextSibling);                    // insert the New node after the targeted Node
</script>

The cloneNode() / importNode() / adoptNode() methods 

SYNTAX :    referenceNode.cloneNode(deepCopy[bool])                                    // creates a new cloned Node and returns it
            document.importNode(node/docFragment, deepCopy[bool])                      // clones a node from another document and returns it as an element of the current document     
            document.adoptNode(node/docFragment)                                       // transfers (removes) a node from another document and returns it as an element of the current document    
                - deepCopy[bool] : (default false) clone children elements or not 
    
    <p id="demo"7> Demo Paragraph </p>
<script>
    var child = document.getElementById('demo');
    var parent = document.body;                                                        // the "body" element is the parent here 
    
    var child_2 = child.cloneNode(true);                                               // clone the Node's children too!
    parent.insertBefore(child_2, child);                                               // the clone inserted before the original element
    parent.insertBefore(child_2, child.nextSibling);                                   // this is how you would insert the new Node after the original element   
<script>
    
    <template id="myTemplate">                                                         // the template element is a "document fragment" not part of the current document 
        <p> elso elem </p>
        <p> masodik elem </p>
    </template>
<script>
    var extDoc = document.getElementById("myTemplate");                                // document fragment
    
// importNode -----------------------
    var nd_1 = extDoc.content.firstElementChild;                                       // select the targeted node from the source document
    var copyNode = document.importNode(nd_1, true);                                    // node cloned and iported to the current document 
    extDoc.parentElement.insertBefore(copyNode, extDoc);                               // insert the imported node
    
// adoptNode ------------------------
    var nd_2 = extDoc.content.lastElementChild;                                        // select the targeted node from the source document     
    var adoptedNode = document.adoptNode(nd_2);                                        // node removed from the source document and imported to the current document (no deep copy argument)
    extDoc.parentElement.insertBefore(adoptedNode, extDoc);                            // insert the imported noe
</script>

importNode / adoptNode TEST

CODE : 
    <template id="myTemplate">
        <p> first element </p>
        <p> second element </p>
    </template>
    <div id="paste"></div>
<script>
    var template_ = document.getElementById("myTemplate");          // template is a "document fragment" = not part of this document!
    
// importNode -----------    
    var foreignNode_1 = template_.content.firstElementChild;
    var importNode_1 = document.importNode(foreignNode_1, true);    // we imported the node to the current document (this is a clone)
    document.getElementById("paste").appendChild(importNode_1);
    
// adoptNode ------------
    var foreignNode_2 = template_.content.children[1];
    var importNode_2 = document.adoptNode(foreignNode_2, true);     // we adopted the node from the original (node removed and stored) 
    document.getElementById("paste").appendChild(importNode_2);
</script>

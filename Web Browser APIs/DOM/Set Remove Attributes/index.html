Description and Demonstration :

- the relation between an element and its attributes in JavaScript is represented as object(element) and its properties(attributes) but the delete operator stil cannot delete element Attributes

- attribute Node methods are sort of deprecated because takes extra work and code to deal with them! 

SYNTAX :    element.id = "value"                                  // get or set element's id attribute 
            element.className                                     // gets or sets the class attribute of the element (used instead of the class attribute to avoid conflict with the "class" keyword)

            element.setAttribute("attributeName", "value")        // create or overwrite arguments               
            element.getAttribute("attributeName")                 // gets the specified attribute's value        
            element.removeAttribute("attributeName")              // removes the specified attribute and returns it      
            element.hasAttribute("attributeName")                 // returns a boolean value indicating if the element has the specified attribute 
            element.hasAttributes()                               // returns a boolean value indicating if the element has any attributes
            
            document.createAttribute("newAttribute")              // creates an attribute Node
            element.setAttributeNode(attributeNode)               // sets an attribute Node to an element
            element.getAttributeNode("attributeName")             // gets an attribute Node and returns it
            element.removeAttributeNode(attributeNode)            // remove an attribute Node
            
            element.attributes                                    // returns a "NamedNodeMap" (Attribute Node List) of the element's attributes
            namedNodeMap[index/"attrKey"]                         // returns the targeted attribute from a "NamedNodeMap"
            namedNodeMap.getNamedItem("attrKey")                  // returns the targeted attribute node from a "NamedNodeMap"  
            namedNodeMap.setNamedItem(attrNode)                   // sets (adds) an attribute node to the "NamedNodeMap"
            namedNodeMap.removeNamedItem("attrKey")               // removes the targeted attribute node from the "NamedNodeMap" and returns it

the classList property (DOMTokenList)

the length property and the add() / contains() / item() / remove() / replace() / toggle() methods 

SYNTAX :    element.classList                                                   // returns the DOMTokenList object which contains the element's class attribute data (read only property)
            
            domTokenList.length                                                 // returns the number of class id's in the Class List
            
            domTokenList.add(classId:str, ...)                                  // add new class id(s) to the Class List
            domTokenList.intem(index)                                           // returns the targeted classId in the sequence 
            domTokenList.contains(classId:str)                                  // returns a boolean indicating if the specified class id is present in the Class List   
            
            domTokenList.remove(classId:str, ...)                               // remove the specified classId(s) from the Class List   
            domTokenList.replace(oldClassId:str, newClassId:str)                // replaces the targeted classId with the passed new classId 
            
            domTokenList.toggle(classId:str, force:bol)                         // toggles the specified classId every time it's called and returns a boolean indicating of the current toggle state   
                - force:bol : defines the permanent state of the of the classId (true = toggled on | false = toggled off)   
    
    <p class="one"> myParagraph </p>
<script>
    var clsList = document.getElementsByClassName('one')[0].classList;          // returns the DOMTokenList (classList) of the element 
    
// length / add() / item() / contains() ----------------------------------------
    clsList.add('two', 'three');                                                // we added two new classIds to the element       
    clsList.length;                                                             // -> 3         // number of classIds in the classList 
    clsList.item(1);                                                            // -> 'two'     // returns the second classId in the sequence 
    clsList.contains('three');                                                  // -> true      // classId is present in the classList   
    
// remove() / replace() --------------------------------------------------------
    clsList.remove('three');                                                    // removes the targeted classId from the classList   
    clsList.replace('one', 'first');                                            // replaces the 'one' classId with the 'first' classId in the classList  
    
// toggle() --------------------------------------------------------------------
  // every time the toggle() is called will toggle the specified classId  
    clsList.toggle('newCls');                                                   // -> true      // toggles on the 'newCls' classId 
    clsList.toggle('newCls');                                                   // -> false     // toggles off the 'newCls' classId  
    
    clsList.toggle('newCls', false);                                            // -> false     // -! this forces the classId to stay toggled off
    clsList.toggle('newCls', true);                                             // -> true   
</script>

the entries() / keys() / values() / forEach() methods 

SYNTAX :    domTokenList.eitries()                                              // returns an iterable object which contains the key/value pairs of the classList   
            domTokenList.keys()                                                 // returns an iterable object which contains the keys of the classList     
            domTokenList.values()                                               // returns an iterable object which contains the values of the classList   
            
            domTokenList.forEach(fn(val, index, domTokenList), thisVal)         // runs the callback function for each item in the classList  
                - thisVal : defines the this value for the callback function 
    
    var clsList = document.getElementsByClassName('one')[0].classList;          // returns the DOMTokenList (classList) of the element  
        
// entries() -------------------------------------------------------------------
    var entryCollect = clsList.entries();                                       // iteralbe object contains the key/value pairs of the classList   
    for(i of entryCollect) { i }                                                // -> [0, "one"] /Â [1, "two"] / [2, "three"]
    
// keys() ----------------------------------------------------------------------
    var keysCollect = clsList.keys();                                           // iterable object contains the keys of the classList items
    for(i of keysCollect) { i }                                                 // -> 0 / 1 / 2
    
// values() --------------------------------------------------------------------
    var valuesCollect = clsList.values();                                       // iterable object contains the values of the classList items 
    or(i of valuesCollect) { i }                                                // -> 'one' / 'two' / 'three'  
    
// forEach() -------------------------------------------------------------------
    clsList.forEach( function(val, ind, clsL){                                  // run a pass for every classList item 
        ind;                                                                    // -> 0 / 1 / 2
        val;                                                                    // -> 'one' / 'two' / 'three'
        clsl;                                                                   // -> clsList / clsList / clsList       // returns the DOMTokenList itself
    });

<script>
    var clsList2 = document.getElementsByClassName('un')[0].classList;
    
// entries() ---------------------------------------------------
    var clsListEntries = clsList2.entries();                    // iterable object (contains the key/value pairs)
    
    var entry_1 = clsListEntries.next();                        // -> {value:[0, 'un'], done:false}
    var entry_2 = clsListEntries.next();                        // -> {value:[1, 'deux'], done:false}
    var entry_3 = clsListEntries.next();                        // -> {value:[2, 'trois'], done:false}
    var entry_4 = clsListEntries.next();                        // -> {value:undefined, done:true}          // end of iterable object

    var clsListKeys = clsList2.keys();                          // iterable object (contains the keys)
    
    var key_1 = clsListKeys.next();                             // -> {value:0, done:false}
    var key_2 = clsListKeys.next();                             // -> {value:1, done:false}
    var key_3 = clsListKeys.next();                             // -> {value:2, done:false}
    var key_4 = clsListKeys.next();                             // -> {value:undefined, done:true}
    
// values() ----------------------------------------------------
    var clsListValues = clsList2.values();                      // iterable object (contains the actual values)
    var collectValues = [];
    
    for(i of clsListValues) {
        collectValues.push(i);                                  // -> ['un', 'deux', 'trois']
